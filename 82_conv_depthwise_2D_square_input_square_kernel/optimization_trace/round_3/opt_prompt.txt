@triton.jit
def _dwconv2d_kernel_coalesced(
    x_ptr, w_ptr, b_ptr, y_ptr,
    N, C, H, W,
    H_OUT, W_OUT,
    X_STRIDE_N, X_STRIDE_C, X_STRIDE_H, X_STRIDE_W,
    W_STRIDE_C, W_STRIDE_KH, W_STRIDE_KW,
    Y_STRIDE_N, Y_STRIDE_C, Y_STRIDE_H, Y_STRIDE_W,
    K: tl.constexpr,
    STRIDE: tl.constexpr,
    PADDING: tl.constexpr,
    HAS_BIAS: tl.constexpr,
    BLOCK_H: tl.constexpr,
    BLOCK_W: tl.constexpr,
):
    """Optimized depthwise conv with coalesced memory access and data reuse."""
    pid_w = tl.program_id(0)
    pid_h = tl.program_id(1)
    pid_nc = tl.program_id(2)

    c = pid_nc % C
    n = pid_nc // C

    # Output tile coordinates - BLOCK_W should be warp-aligned (32)
    offs_w = pid_w * BLOCK_W + tl.arange(0, BLOCK_W)
    offs_h = pid_h * BLOCK_H + tl.arange(0, BLOCK_H)
    
    mask_w = offs_w < W_OUT
    mask_h = offs_h < H_OUT

    # Input base coordinates (for stride=1, padding=0 case optimized)
    ix_base = offs_w * STRIDE - PADDING
    iy_base = offs_h * STRIDE - PADDING

    x_base = x_ptr + n * X_STRIDE_N + c * X_STRIDE_C
    w_base = w_ptr + c * W_STRIDE_C

    # Initialize accumulator
    acc = tl.zeros([BLOCK_H, BLOCK_W], dtype=tl.float32)

    # Load all weights for this channel upfront (small K, fits in registers)
    for ky in range(K):
        iy = iy_base[:, None] + ky
        valid_y = (iy >= 0) & (iy < H)
        
        for kx in range(K):
            ix = ix_base[None, :] + kx
            valid_x = (ix >= 0) & (ix < W)
            
            # Combined validity mask
            m = mask_h[:, None] & mask_w[None, :] & valid_y & valid_x
            
            # Coalesced load - consecutive threads load consecutive memory locations
            x_ptrs = x_base + iy * X_STRIDE_H + ix * X_STRIDE_W
            x_vals = tl.load(x_ptrs, mask=m, other=0.0).to(tl.float32)
            
            # Weight for this kernel position
            w_val = tl.load(w_base + ky * W_STRIDE_KH + kx * W_STRIDE_KW).to(tl.float32)
            
            acc += x_vals * w_val

    if HAS_BIAS:
        b_val = tl.load(b_ptr + c).to(tl.float32)
        acc += b_val

    # Coalesced store
    y_base = y_ptr + n * Y_STRIDE_N + c * Y_STRIDE_C
    y_ptrs = y_base + offs_h[:, None] * Y_STRIDE_H + offs_w[None, :] * Y_STRIDE_W
    out_mask = mask_h[:, None] & mask_w[None, :]
    tl.store(y_ptrs, acc.to(tl.bfloat16), mask=out_mask)
