import torch
import triton
import triton.language as tl


@triton.jit
def _dwconv2d_kernel_coalesced(
    x_ptr, w_ptr, b_ptr, y_ptr,
    N, C, H, W,
    H_OUT, W_OUT,
    X_STRIDE_N, X_STRIDE_C, X_STRIDE_H, X_STRIDE_W,
    W_STRIDE_C, W_STRIDE_KH, W_STRIDE_KW,
    Y_STRIDE_N, Y_STRIDE_C, Y_STRIDE_H, Y_STRIDE_W,
    K: tl.constexpr,
    STRIDE: tl.constexpr,
    PADDING: tl.constexpr,
    HAS_BIAS: tl.constexpr,
    BLOCK_H: tl.constexpr,
    BLOCK_W: tl.constexpr,
):
    """Optimized depthwise conv with coalesced memory access and data reuse."""
    pid_w = tl.program_id(0)
    pid_h = tl.program_id(1)
    pid_nc = tl.program_id(2)

    c = pid_nc % C
    n = pid_nc // C

    # Output tile coordinates - BLOCK_W should be warp-aligned (32)
    offs_w = pid_w * BLOCK_W + tl.arange(0, BLOCK_W)
    offs_h = pid_h * BLOCK_H + tl.arange(0, BLOCK_H)
    
    mask_w = offs_w < W_OUT
    mask_h = offs_h < H_OUT

    # Input base coordinates (for stride=1, padding=0 case optimized)
    ix_base = offs_w * STRIDE - PADDING
    iy_base = offs_h * STRIDE - PADDING

    x_base = x_ptr + n * X_STRIDE_N + c * X_STRIDE_C
    w_base = w_ptr + c * W_STRIDE_C

    # Initialize accumulator
    acc = tl.zeros([BLOCK_H, BLOCK_W], dtype=tl.float32)

    # Load all weights for this channel upfront (small K, fits in registers)
    for ky in range(K):
        iy = iy_base[:, None] + ky
        valid_y = (iy >= 0) & (iy < H)
        
        for kx in range(K):
            ix = ix_base[None, :] + kx
            valid_x = (ix >= 0) & (ix < W)
            
            # Combined validity mask
            m = mask_h[:, None] & mask_w[None, :] & valid_y & valid_x
            
            # Coalesced load - consecutive threads load consecutive memory locations
            x_ptrs = x_base + iy * X_STRIDE_H + ix * X_STRIDE_W
            x_vals = tl.load(x_ptrs, mask=m, other=0.0).to(tl.float32)
            
            # Weight for this kernel position
            w_val = tl.load(w_base + ky * W_STRIDE_KH + kx * W_STRIDE_KW).to(tl.float32)
            
            acc += x_vals * w_val

    if HAS_BIAS:
        b_val = tl.load(b_ptr + c).to(tl.float32)
        acc += b_val

    # Coalesced store
    y_base = y_ptr + n * Y_STRIDE_N + c * Y_STRIDE_C
    y_ptrs = y_base + offs_h[:, None] * Y_STRIDE_H + offs_w[None, :] * Y_STRIDE_W
    out_mask = mask_h[:, None] & mask_w[None, :]
    tl.store(y_ptrs, acc.to(tl.bfloat16), mask=out_mask)


@triton.jit
def _dwconv2d_kernel_k3_fast(
    x_ptr, w_ptr, b_ptr, y_ptr,
    N, C, H, W,
    H_OUT, W_OUT,
    X_STRIDE_N, X_STRIDE_C, X_STRIDE_H, X_STRIDE_W,
    W_STRIDE_C, W_STRIDE_KH, W_STRIDE_KW,
    Y_STRIDE_N, Y_STRIDE_C, Y_STRIDE_H, Y_STRIDE_W,
    STRIDE: tl.constexpr,
    PADDING: tl.constexpr,
    HAS_BIAS: tl.constexpr,
    BLOCK_H: tl.constexpr,
    BLOCK_W: tl.constexpr,
):
    """Specialized 3x3 kernel with unrolled loops and optimized memory access."""
    pid_w = tl.program_id(0)
    pid_h = tl.program_id(1)
    pid_nc = tl.program_id(2)

    c = pid_nc % C
    n = pid_nc // C

    offs_w = pid_w * BLOCK_W + tl.arange(0, BLOCK_W)
    offs_h = pid_h * BLOCK_H + tl.arange(0, BLOCK_H)
    
    mask_w = offs_w < W_OUT
    mask_h = offs_h < H_OUT

    ix_base = offs_w * STRIDE - PADDING
    iy_base = offs_h * STRIDE - PADDING

    x_base = x_ptr + n * X_STRIDE_N + c * X_STRIDE_C
    w_base = w_ptr + c * W_STRIDE_C

    # Load all 9 weights upfront
    w00 = tl.load(w_base).to(tl.float32)
    w01 = tl.load(w_base + W_STRIDE_KW).to(tl.float32)
    w02 = tl.load(w_base + 2 * W_STRIDE_KW).to(tl.float32)
    w10 = tl.load(w_base + W_STRIDE_KH).to(tl.float32)
    w11 = tl.load(w_base + W_STRIDE_KH + W_STRIDE_KW).to(tl.float32)
    w12 = tl.load(w_base + W_STRIDE_KH + 2 * W_STRIDE_KW).to(tl.float32)
    w20 = tl.load(w_base + 2 * W_STRIDE_KH).to(tl.float32)
    w21 = tl.load(w_base + 2 * W_STRIDE_KH + W_STRIDE_KW).to(tl.float32)
    w22 = tl.load(w_base + 2 * W_STRIDE_KH + 2 * W_STRIDE_KW).to(tl.float32)

    acc = tl.zeros([BLOCK_H, BLOCK_W], dtype=tl.float32)

    # Process each kernel row
    for ky in tl.static_range(3):
        iy = iy_base[:, None] + ky
        valid_y = (iy >= 0) & (iy < H)
        row_base = x_base + iy * X_STRIDE_H

            if ky == 0:

                elif kx == 1:
                    w_val = w01
                else:
                    w_val = w02
            elif ky == 1:
                if kx == 0:
                    w_val = w10
                elif kx == 1:
                    w_val = w11
                else:
                    w_val = w12
            else:
                if kx == 0:
                    w_val = w20
                elif kx == 1:
                    w_val = w21
                else:
                    w_val = w22
            
            acc += x_val * w_val

    if HAS_BIAS:
        b_val = tl.load(b_ptr + c).to(tl.float32)
        acc += b_val

    y_base = y_ptr + n * Y_STRIDE_N + c * Y_STRIDE_C
    y_ptrs = y_base + offs_h[:, None] * Y_STRIDE_H + offs_w[None, :] * Y_STRIDE_W

def kernel_function(input, weight, bias=None, stride=1, padding=0):
    assert input.is_cuda and weight.is_cuda
    device = input.device

    stride_h = stride_w = int(stride) if isinstance(stride, int) else (int(stride[0]), int(stride[1]))[1]
    pad_h = pad_w = int(padding) if isinstance(padding, int) else (int(padding[0]), int(padding[1]))[1]

    N, C, H, W = input.shape

    if weight.dim() == 4:
        K = weight.shape[2]
        w3 = weight.view(C, K, K).contiguous()
    else:
        K = weight.shape[1]
        w3 = weight.contiguous()

    H_out = (H + 2 * pad_h - K) // stride_h + 1
    W_out = (W + 2 * pad_w - K) // stride_w + 1

    y = torch.empty((N, C, H_out, W_out), device=device, dtype=input.dtype)

    XsN, XsC, XsH, XsW = input.stride()
    WsC, WsKH, WsKW = w3.stride()
    YsN, YsC, YsH, YsW = y.stride()

    has_bias = bias is not None

    # Optimized tile sizes for H100 - BLOCK_W=128 for better coalescing
    BLOCK_H = 8
    BLOCK_W = 128

    def grid(meta):
        return (
            triton.cdiv(W_out, meta["BLOCK_W"]),
            triton.cdiv(H_out, meta["BLOCK_H"]),
            N * C
        )

    if K == 3:
        _dwconv2d_kernel_k3_fast[grid](
            input, w3, bias if has_bias else y, y,
            N, C, H, W,
            H_out, W_out,
            XsN, XsC, XsH, XsW,
            WsC, WsKH, WsKW,
            YsN, YsC, YsH, YsW,
            STRIDE=stride_w,
            PADDING=pad_w,
            HAS_BIAS=has_bias,
            BLOCK_H=BLOCK_H,
            BLOCK_W=BLOCK_W,
            num_warps=8,
            num_stages=4,
        )
    else:
        _dwconv2d_kernel_coalesced[grid](
            input, w3, bias if has_bias else y, y,
            N, C, H, W,
            H_out, W_out,
            XsN, XsC, XsH, XsW,
            WsC, WsKH, WsKW,
            YsN, YsC, YsH, YsW,
            K=K,
            STRIDE=stride_w,
            PADDING=pad_w,
            HAS_BIAS=has_bias,
            BLOCK_H=BLOCK_H,
            BLOCK_W=BLOCK_W,
            num_warps=8,
            num_stages=4,
        )

    return y
